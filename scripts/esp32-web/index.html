on SensorNev do
  if [SensorNev] < 18 and [hour] >= 6
    on Eszkoz
  else

  endif
endon

on timer 1 do
  off Pump
endon

on boot do
  set var1 0
endon

on time 18:00 do
  on Light
endon

// Parancsok: on/off/toggle Eszkoz
// Timer: timerSet N mp, timerCancel N
// Valtozo: set var1 ertek
// Ertekek: [SensorNev] [var1] [hour]
//          [minute] [uptime]
// Operatorok: < > <= >= = !=
// Logika: and, or, not
